Q6:

create or replace procedure tab_names as

cursor cur is select table_name from all_tables;

begin
for rec in cur loop
dbms_output.put_line(rec.table_name);
end loop;
end;
/

set serveroutput on;

begin
tab_names();
end;
/

create or replace procedure PS_GET_TABLES_OF_USER (arg_user varchar2) is
       cursor C1 is select object_name,created from dba_objects where (object_type='TABLE') and (owner= arg_user) order by object_name;
       begin
	dbms_output.enable;
	for l in C1
	loop
	DBMS_OUTPUT.PUT_LINE('nom de tabe: '|| rpad(l.object_name,25,' ') ||'date de creation:   '|| l.created);
	end loop;	
	end ;
	/
--executer la procedure:
system>execute PS_GET_TABLES_OF_USER ('HR');
